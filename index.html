<script>

document.getElementById('allocationForm').addEventListener('submit', function(event) {

  event.preventDefault();


  const teamA = parseFloat(document.getElementById('teamA').value) || 0;

  const teamB = parseFloat(document.getElementById('teamB').value) || 0;

  const teamC = parseFloat(document.getElementById('teamC').value) || 0;


  const total = teamA + teamB + teamC;


  const resultDiv = document.getElementById('result');


  if (Math.abs(total - 1500) < 0.01) {

    resultDiv.innerHTML = `<span class="success">✅ Thank you! Your allocation has been recorded.</span>`;

    console.log({teamA, teamB, teamC});

  } else {

    resultDiv.innerHTML = `<span class="error">❌ Your total is ${total.toFixed(1)} points. Please adjust to exactly 1,500 points.</span>`;

  }

});
